
coa_revsocialism_trigger = {
    has_law = law_type:law_council_republic
    coa_autocracy_trigger = no
	OR = {
		ig:ig_trade_unions = {
			is_in_government = yes
		}
		any_interest_group = {
			is_in_government = yes
			has_ideology = ideology_proletarian
		}
	}
}

coa_anarchist_trigger = {
    has_law = law_type:law_anarchy
    coa_socialism_trigger = yes
}

coa_ML_trigger = {
    has_law = law_type:law_command_economy
    has_law = law_type:law_council_republic
}

coa_sythesis_trigger = {
    coa_socialism_trigger = yes
	coa_ML_trigger = yes
}

coa_independent_trigger = {
	NOT = {exists = scope:overlord}
}

# Government

coa_def_revsocialism_flag_trigger = {
	exists = scope:actor
	scope:actor = { coa_revsocialism_trigger = yes }
}

coa_def_anarchist_flag_trigger = {
	exists = scope:actor
	scope:actor = { coa_anarchist_trigger = yes }
}

coa_def_ML_flag_trigger = {
	exists = scope:actor
	scope:actor = { coa_ML_trigger = yes }
}

coa_def_sythesis_flag_trigger = {
	exists = scope:actor
	scope:actor = { coa_sythesis_trigger = yes }
}
